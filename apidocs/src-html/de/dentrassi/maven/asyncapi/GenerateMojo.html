017package de.dentrassi.maven.asyncapi;
018
019import java.io.File;
020import java.io.InputStream;
021import java.nio.charset.Charset;
022import java.nio.file.Files;
023import java.util.Arrays;
024import java.util.HashSet;
025import java.util.Set;
026
027import org.apache.maven.plugin.AbstractMojo;
028import org.apache.maven.plugin.MojoExecutionException;
029import org.apache.maven.plugin.MojoFailureException;
030import org.apache.maven.plugins.annotations.Component;
031import org.apache.maven.plugins.annotations.LifecyclePhase;
032import org.apache.maven.plugins.annotations.Mojo;
033import org.apache.maven.plugins.annotations.Parameter;
034import org.apache.maven.project.MavenProject;
035import org.sonatype.plexus.build.incremental.BuildContext;
036
037import de.dentrassi.asyncapi.AsyncApi;
038import de.dentrassi.asyncapi.ValidationException;
039import de.dentrassi.asyncapi.Validator.Marker;
040import de.dentrassi.asyncapi.generator.java.Generator;
041import de.dentrassi.asyncapi.generator.java.gson.GsonGeneratorExtension;
042import de.dentrassi.asyncapi.generator.java.jms.JmsGeneratorExtension;
043import de.dentrassi.asyncapi.internal.parser.YamlParser;
044
045/**
046 * Generate sources from AsyncAPI definition
047 */
048@Mojo(name = "generate", defaultPhase = LifecyclePhase.GENERATE_SOURCES, requiresProject = true, threadSafe = false)
049public class GenerateMojo extends AbstractMojo {
050
051    @Parameter(property = "asyncapi.definition", required = true, defaultValue = "${project.basedir}/src/main/asyncapi.yaml")
052    private File definitionFile;
053
054    @Parameter(property = "asyncapi.ignoreMissingDefinition", required = false, defaultValue = "true")
055    private boolean ignoreMissingDefinition = true;
056
057    /**
058     * Allows to skip the whole generation step
059     */
060    @Parameter(property = "asyncapi.skip", required = false, defaultValue = "false")
061    private boolean skip;
062
063    /**
064     * The character set to use for generated resources
065     */
066    @Parameter(property = "asyncapi.generator.charset", required = true, defaultValue = "${project.build.sourceEncoding}")
067    private String characterSet;
068
069    /**
070     * The Java base package for generated
071     * <p>
072     * Will use the {@code baseTopic} prefix from the API model is unspecified
073     * </p>
074     */
075    @Parameter(property = "asyncapi.generator.packageBase", required = false)
076    private String packageBase;
077
078    /**
079     * The path to generate the sources to
080     */
081    @Parameter(required = true, defaultValue = "${project.build.directory}/generated-sources/asyncapi")
082    private File targetPath;
083
084    @Parameter(property = "project", readonly = true, required = true)
085    protected MavenProject project;
086
087    @Parameter
088    private Set<String> extensions;
089
090    @Component
091    private BuildContext buildContext;
092
093    public void setDefinitionFile(final File definitionFile) {
094        this.definitionFile = definitionFile;
095    }
096
097    public void setIgnoreMissingDefinition(final boolean ignoreMissingDefinition) {
098        this.ignoreMissingDefinition = ignoreMissingDefinition;
099    }
100
101    public void setSkip(final boolean skip) {
102        this.skip = skip;
103    }
104
105    public void setCharacterSet(final String characterSet) {
106        this.characterSet = characterSet;
107    }
108
109    public void setPackageBase(final String packageBase) {
110        this.packageBase = packageBase;
111    }
112
113    public void setProject(final MavenProject project) {
114        this.project = project;
115    }
116
117    public void setBuildContext(final BuildContext buildContext) {
118        this.buildContext = buildContext;
119    }
120
121    public void setExtensions(final Set<String> extensions) {
122        this.extensions = extensions;
123    }
124
125    @Override
126    public void execute() throws MojoExecutionException, MojoFailureException {
127        if (this.skip) {
128            return;
129        }
130
131        if (!this.definitionFile.exists() && this.ignoreMissingDefinition) {
132            getLog().debug(String.format("Skipping generation. Definition file '%s' is missing and we ignore this", this.definitionFile));
133            return;
134        }
135
136        // load
137
138        getLog().info(String.format("Reading definition: %s", this.definitionFile));
139
140        final AsyncApi api;
141
142        try (final InputStream in = Files.newInputStream(this.definitionFile.toPath())) {
143
144            api = new YamlParser(in).parse();
145
146        } catch (final ValidationException e) {
147
148            getLog().error("Model validation failed: ");
149            for (final Marker marker : e.getMarkers()) {
150                getLog().error(marker.toString());
151            }
152            throw new MojoFailureException("Invalid API model", e);
153
154        } catch (final Exception e) {
155
156            throw new MojoFailureException("Failed to read definition", e);
157
158        }
159
160        // extract info
161
162        String title = api.getInformation().getTitle();
163        if (title == null) {
164            title = "unknown";
165        }
166        String version = api.getInformation().getVersion();
167        if (version == null) {
168            version = "unknown";
169        }
170
171        // generate
172
173        getLog().info(String.format("Generating API: %s:%s", title, version));
174        getLog().debug(String.format("    Output: %s", this.targetPath));
175
176        final Generator.Builder generator = Generator.newBuilder();
177        generator.characterSet(Charset.forName(this.characterSet));
178        generator.basePackage(this.packageBase);
179        generator.targetPath(this.targetPath.toPath());
180
181        addNamedExtensions(generator);
182
183        try {
184            generator.build(api).generate();
185        } catch (final Exception e) {
186            throw new MojoFailureException("Failed to generate API", e);
187        }
188
189        // add sources
190
191        this.project.addCompileSourceRoot(this.targetPath.getAbsolutePath());
192        this.buildContext.refresh(this.targetPath.getAbsoluteFile());
193    }
194
195    private void addNamedExtensions(final Generator.Builder generator) throws MojoExecutionException {
196
197        if (this.extensions == null) {
198            this.extensions = new HashSet<>(Arrays.asList("jms", "gson"));
199            getLog().info("Using default extensions: " + this.extensions);
200        }
201
202        for (final String extension : this.extensions) {
203            switch (extension.toLowerCase()) {
204            case "jms":
205                generator.addExtension(new JmsGeneratorExtension());
206                break;
207            case "gson":
208                generator.addExtension(new GsonGeneratorExtension());
209                break;
210            default:
211                throw new MojoExecutionException(String.format("Unknown generator extension '%s'", extension));
212            }
213        }
214    }
215
216}
