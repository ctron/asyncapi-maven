017package de.dentrassi.maven.asyncapi;
018
019import java.io.File;
020import java.io.InputStream;
021import java.nio.charset.Charset;
022import java.nio.file.Files;
023import java.util.Arrays;
024import java.util.HashSet;
025import java.util.Set;
026
027import org.apache.maven.plugin.AbstractMojo;
028import org.apache.maven.plugin.MojoExecutionException;
029import org.apache.maven.plugin.MojoFailureException;
030import org.apache.maven.plugins.annotations.Component;
031import org.apache.maven.plugins.annotations.LifecyclePhase;
032import org.apache.maven.plugins.annotations.Mojo;
033import org.apache.maven.plugins.annotations.Parameter;
034import org.apache.maven.project.MavenProject;
035import org.sonatype.plexus.build.incremental.BuildContext;
036
037import de.dentrassi.asyncapi.AsyncApi;
038import de.dentrassi.asyncapi.ValidationException;
039import de.dentrassi.asyncapi.Validator.Marker;
040import de.dentrassi.asyncapi.generator.java.Generator;
041import de.dentrassi.asyncapi.generator.java.gson.GsonGeneratorExtension;
042import de.dentrassi.asyncapi.generator.java.jms.JmsClientGeneratorExtension;
043import de.dentrassi.asyncapi.internal.parser.YamlParser;
044
045/**
046 * Generate sources from AsyncAPI definition
047 */
048@Mojo(name = "generate", defaultPhase = LifecyclePhase.GENERATE_SOURCES, requiresProject = true, threadSafe = false)
049public class GenerateMojo extends AbstractMojo {
050
051    @Parameter(property = "asyncapi.definition", required = true, defaultValue = "${project.basedir}/src/main/asyncapi.yaml")
052    private File definitionFile;
053
054    @Parameter(property = "asyncapi.ignoreMissingDefinition", required = false, defaultValue = "false")
055    private boolean ignoreMissingDefinition;
056
057    @Parameter(property = "asyncapi.skip", required = false, defaultValue = "false")
058    private boolean skip;
059
060    @Parameter(property = "asyncapi.generator.charset", required = true, defaultValue = "${project.build.sourceEncoding}")
061    private String characterSet;
062
063    @Parameter(property = "asyncapi.generator.packageBase", required = false)
064    private String packageBase;
065
066    @Parameter(required = true, defaultValue = "${project.build.directory}/generated-sources/asyncapi")
067    private File targetPath;
068
069    @Parameter(property = "project", readonly = true, required = true)
070    protected MavenProject project;
071
072    @Parameter
073    private Set<String> extensions;
074
075    @Component
076    private BuildContext buildContext;
077
078    public void setDefinitionFile(final File definitionFile) {
079        this.definitionFile = definitionFile;
080    }
081
082    public void setIgnoreMissingDefinition(final boolean ignoreMissingDefinition) {
083        this.ignoreMissingDefinition = ignoreMissingDefinition;
084    }
085
086    public void setSkip(final boolean skip) {
087        this.skip = skip;
088    }
089
090    public void setCharacterSet(final String characterSet) {
091        this.characterSet = characterSet;
092    }
093
094    public void setPackageBase(final String packageBase) {
095        this.packageBase = packageBase;
096    }
097
098    public void setProject(final MavenProject project) {
099        this.project = project;
100    }
101
102    public void setBuildContext(final BuildContext buildContext) {
103        this.buildContext = buildContext;
104    }
105
106    public void setExtensions(final Set<String> extensions) {
107        this.extensions = extensions;
108    }
109
110    @Override
111    public void execute() throws MojoExecutionException, MojoFailureException {
112        if (this.skip) {
113            return;
114        }
115
116        if (!this.definitionFile.exists() && this.ignoreMissingDefinition) {
117            getLog().debug(String.format("Skipping. Definition file '%s' is missing but we ignore this", this.definitionFile));
118            return;
119        }
120
121        // load
122
123        getLog().info(String.format("Reading definition: %s", this.definitionFile));
124
125        final AsyncApi api;
126
127        try (final InputStream in = Files.newInputStream(this.definitionFile.toPath())) {
128
129            api = new YamlParser(in).parse();
130
131        } catch (final ValidationException e) {
132
133            getLog().error("Model validation failed: ");
134            for (final Marker marker : e.getMarkers()) {
135                getLog().error(marker.toString());
136            }
137            throw new MojoFailureException("Invalid API model", e);
138
139        } catch (final Exception e) {
140
141            throw new MojoFailureException("Failed to read definition", e);
142
143        }
144
145        // extract info
146
147        String title = api.getInformation().getTitle();
148        if (title == null) {
149            title = "unknown";
150        }
151        String version = api.getInformation().getVersion();
152        if (version == null) {
153            version = "unknown";
154        }
155
156        // generate
157
158        getLog().info(String.format("Generating API: %s:%s", title, version));
159        getLog().debug(String.format("    Output: %s", this.targetPath));
160
161        final Generator.Builder generator = Generator.newBuilder();
162        generator.characterSet(Charset.forName(this.characterSet));
163        generator.basePackage(this.packageBase);
164        generator.targetPath(this.targetPath.toPath());
165
166        addNamedExtensions(generator);
167
168        try {
169            generator.build(api).generate();
170        } catch (final Exception e) {
171            throw new MojoFailureException("Failed to generate API", e);
172        }
173
174        // add sources
175
176        this.project.addCompileSourceRoot(this.targetPath.getAbsolutePath());
177        this.buildContext.refresh(this.targetPath.getAbsoluteFile());
178    }
179
180    private void addNamedExtensions(final Generator.Builder generator) throws MojoExecutionException {
181
182        if (this.extensions == null) {
183            this.extensions = new HashSet<>(Arrays.asList("jms-client", "gson"));
184            getLog().info("Using default extensions: " + this.extensions);
185        }
186
187        for (final String extension : this.extensions) {
188            switch (extension.toLowerCase()) {
189            case "jms-client":
190                generator.addExtension(new JmsClientGeneratorExtension());
191                break;
192            case "gson":
193                generator.addExtension(new GsonGeneratorExtension());
194                break;
195            default:
196                throw new MojoExecutionException(String.format("Unknown generator extension '%s'", extension));
197            }
198        }
199    }
200
201}
