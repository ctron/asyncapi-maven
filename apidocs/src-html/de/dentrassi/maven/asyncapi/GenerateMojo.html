017package de.dentrassi.maven.asyncapi;
018
019import java.io.File;
020import java.io.InputStream;
021import java.nio.charset.Charset;
022import java.nio.file.Files;
023
024import org.apache.maven.plugin.AbstractMojo;
025import org.apache.maven.plugin.MojoExecutionException;
026import org.apache.maven.plugin.MojoFailureException;
027import org.apache.maven.plugins.annotations.Component;
028import org.apache.maven.plugins.annotations.LifecyclePhase;
029import org.apache.maven.plugins.annotations.Mojo;
030import org.apache.maven.plugins.annotations.Parameter;
031import org.apache.maven.project.MavenProject;
032import org.sonatype.plexus.build.incremental.BuildContext;
033
034import de.dentrassi.asyncapi.AsyncApi;
035import de.dentrassi.asyncapi.generator.java.Generator;
036import de.dentrassi.asyncapi.internal.parser.YamlParser;
037
038/**
039 * Generate sources from AsyncAPI definition
040 */
041@Mojo(name = "generate", defaultPhase = LifecyclePhase.GENERATE_SOURCES, requiresProject = true, threadSafe = false)
042public class GenerateMojo extends AbstractMojo {
043
044    @Parameter(property = "asyncapi.definition", required = true, defaultValue = "${project.basedir}/src/main/asyncapi.yaml")
045    private File definitionFile;
046
047    @Parameter(property = "asyncapi.ignoreMissingDefinition", required = false, defaultValue = "false")
048    private boolean ignoreMissingDefinition;
049
050    @Parameter(property = "asyncapi.skip", required = false, defaultValue = "false")
051    private boolean skip;
052
053    @Parameter(property = "asyncapi.generator.charset", required = true, defaultValue = "${project.build.sourceEncoding}")
054    private String characterSet;
055
056    @Parameter(property = "asyncapi.generator.packageBase", required = false)
057    private String packageBase;
058
059    @Parameter(required = true, defaultValue = "${project.build.directory}/generated-sources/asyncapi")
060    private File targetPath;
061
062    @Parameter(property = "project", readonly = true, required = true)
063    protected MavenProject project;
064
065    @Component
066    private BuildContext buildContext;
067
068    public void setDefinitionFile(final File definitionFile) {
069        this.definitionFile = definitionFile;
070    }
071
072    public void setIgnoreMissingDefinition(final boolean ignoreMissingDefinition) {
073        this.ignoreMissingDefinition = ignoreMissingDefinition;
074    }
075
076    public void setSkip(final boolean skip) {
077        this.skip = skip;
078    }
079
080    public void setCharacterSet(final String characterSet) {
081        this.characterSet = characterSet;
082    }
083
084    public void setPackageBase(final String packageBase) {
085        this.packageBase = packageBase;
086    }
087
088    public void setProject(final MavenProject project) {
089        this.project = project;
090    }
091
092    public void setBuildContext(final BuildContext buildContext) {
093        this.buildContext = buildContext;
094    }
095
096    @Override
097    public void execute() throws MojoExecutionException, MojoFailureException {
098        if (this.skip) {
099            return;
100        }
101
102        if (!this.definitionFile.exists() && this.ignoreMissingDefinition) {
103            getLog().debug(String.format("Skipping. Definition file '%s' is missing but we ignore this", this.definitionFile));
104            return;
105        }
106
107        // load
108
109        getLog().info(String.format("Reading definition: %s", this.definitionFile));
110
111        final AsyncApi api;
112        try (final InputStream in = Files.newInputStream(this.definitionFile.toPath())) {
113            api = new YamlParser(in).parse();
114        } catch (final Exception e) {
115            throw new MojoFailureException("Failed to read definition", e);
116        }
117
118        // extract info
119
120        String title = api.getInformation().getTitle();
121        if (title == null) {
122            title = "unknown";
123        }
124        String version = api.getInformation().getVersion();
125        if (version == null) {
126            version = "unknown";
127        }
128
129        // generate
130
131        getLog().info(String.format("Generating API: %s:%s", title, version));
132        getLog().debug(String.format("    Output: %s", this.targetPath));
133
134        final Generator generator = new Generator(api);
135        generator.characterSet(Charset.forName(this.characterSet));
136        generator.basePackage(this.packageBase);
137        generator.target(this.targetPath.toPath());
138        try {
139            generator.generate();
140        } catch (final Exception e) {
141            throw new MojoFailureException("Failed to generate API", e);
142        }
143
144        // add sources
145
146        this.project.addCompileSourceRoot(this.targetPath.getAbsolutePath());
147        this.buildContext.refresh(this.targetPath.getAbsoluteFile());
148    }
149
150}
